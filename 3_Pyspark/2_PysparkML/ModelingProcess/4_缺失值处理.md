def drop_distinct_count(df, thresh=32, _id='id'):
    strCol = [i for i, j in df.dtypes if j=='string' and i!=_id]
    ls = []
    for i in strCol:
        ls.append((df.select(i).distinct().count(), i))
    return df.drop(*[j for i, j in ls if i > thresh])
